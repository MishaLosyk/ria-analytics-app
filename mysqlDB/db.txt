`CREATE TABLE users (
        user_id INT(8) unsigned UNIQUE AUTO_INCREMENT NOT NULL,
        email VARCHAR(35) UNIQUE NOT NULL,
        password VARCHAR(35)  NOT NULL, 
        name VARCHAR(35) NOT NULL,
        surname VARCHAR(35) NOT NULL,
        tables VARCHAR(200) NOT NULL,
        api_key VARCHAR(30) DEFAULT(""),
        PRIMARY KEY (user_id),
        KEY name_idx (name),
        KEY surname_idx (surname),
        KEY tables_idx (tables)
        ) 
        Engine= InnoDb default charset=utf8;
        
        
        
        `CREATE TABLE logs (
		log_id INT(8) unsigned UNIQUE AUTO_INCREMENT NOT NULL,
		user_id INT(8) unsigned NOT NULL,
		date DATETIME NOT NULL,
		query TEXT NOT NULL,
		PRIMARY KEY (log_id),
		KEY user_id_idx (user_id),
		KEY date_idx (date),
		CONSTRAINT user_id_fk FOREIGN KEY (user_id)  REFERENCES users (user_id) ON DELETE CASCADE ON UPDATE CASCADE
	) 
	Engine= InnoDb default charset=utf8`


	`CREATE TABLE queries (
		query_id INT(8) unsigned UNIQUE AUTO_INCREMENT  NOT NULL,
		query_name TEXT NOT NULL,
		query TEXT NOT NULL,
		user_id INT(8) unsigned NOT NULL,
		PRIMARY KEY (query_id),
		KEY user_id_idx (user_id),
		CONSTRAINT user_id_queries_fk FOREIGN KEY (user_id) REFERENCES users (user_id) ON DELETE CASCADE ON UPDATE CASCADE
	)
	Engine= InnoDb default charset=utf8;`
